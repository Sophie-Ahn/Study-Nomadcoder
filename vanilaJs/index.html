variable : 중복 선언도 되고 재할당도 가능함 / let : 중복선언은 안되지만 재할당 가능 / constant : 중복선언도 재할당도 모두 안됨

console.log(name); (값이 seuli로 나옴)
name = "seulkiAhn"; (let 변수명은 변경하지 않았지만 내용은 변경 할 수 있음)
console.log(name); (값이 seulkiAhn으로 나옴)

console.log(something); (undefined로 나옴)

array배열

변수 안에 []를 이용해서 string이나 num을 배열해 줄 수 있음

const array = ["mon", "tue", "wed", "thu", "fri", "sat", "sun"];
console.log(array); (length가 7)
console.log(array[0]); (배열중의 몇번째를 선택하고 싶을때)
array.push("day"); (push는 array에 항목을 하나 더 넣어줄 수 있음)
console.log(array); (array라는 배열 안에 day라는 값이 추가됨)

object객체

jSON방식으로 사용을 많이 함

const player = {
name: 'seulki',
points: 10000,
total: 20000
};
console.log(player);
console.log(player.name); // 쓰는 방식이 여러가지 있음
console.log(player["name"]); // 쓰는 방식이 여러가지 있음
player.name = 'seulkiAhn'; // object안에 name의 데이터를 수정한것
player.addData = 'employee'; // object안에 데이터를 추가 할 수도 있음
player.points = player.points + 10000; // object에 이미 있는 데이터를 받아와 추가 할 수도 있음
console.log(player); // 최종 'player'란 object 데이터가 바뀐 것을 확인 할 수 있음

function함수

function basic(){
console.log("hello basic!");
}
basic(); // hello basic이 나옴

function example(name){ //example이라는 데이터안에 name이란 변수variable이 생김
console.log(name); // name이란 변수를 console.log에 넣음
}
example("seulki"); // name이란 변수를 넣음으로서 example이란 함수안에 데이터를 넣어 호출하면 그대로 나옴
example("gyunghyun");

function examAge(name, age){
console.log(name, age);
}
examAge("seulki", 29); // 변수명을 여러개 넣을 수도 있음

// array를 이용해서 함수를 만듬
const nameTag = ["seulki", "gyunghyun", "end"]
function hello(nameTag){
console.log("hello " + nameTag);
};
hello(nameTag[0]); // array배열을 이용해서 슬라이드를 이용할때 사용하기도 함

// object를 이용해서 함수를 만듬
const objName = {
name: "seulki",
newName: function nameList(newNameList){
console.log("hello!" + newNameList);
}
};
objName.newName("gyunghyun"); // 객체안에 'newName'이라는 데이터값의 함수를 들고옴

return

function을 사용할때 결과값을 원하지 않을때가 많지만 만약 결과값을 원할때 return을 이용해서 function 밖과 연결을 하기도 함

const age = 95;
function calcKorAge(ageOfFore){
return ageOfFore + 2; // function안에서 return의 'ageOfFore + 2;'를 실행하면은
};

const krAge = calcKorAge(age); // 밖에서 function을 실행했을때 'krAge'를 작동하게 만듬/'krAge'는 return 반환값임
console.log(krAge); // return없이 단순히 console.log를 실행했을땐 'undefined'가 뜸. 즉, 값을 정의할 수 없음

if조건문

parseInt() // string으로 쓰여진 숫자를 number 숫자로 변경해줌

const age1 = parseInt(prompt("write number"));
console.log(typeof (age1)); // typeof는 타입을 확인하는 방법

isNaN() // number인지 아닌지를 boolean으로 확인해줌

const age2 = parseInt(prompt("write number"));
console.log(isNaN(age2)); // false가 나오면 숫자라는뜻 true가 나오면 문자라는 뜻
