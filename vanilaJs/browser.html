getElementsByClassName = class를 선택

querySelectorAll = css selector로서 어떤 class나 id안에 태그를 가져올 수 있는데 해당되는 모든 것을 가져 올 수 있음
getElementsByTagName = div, h1, p 등등 태그 자체를 가져 오는 것

자바스크립트로 css 바꾸기

if문 써서 폰트 컬러 변경

const color = document.querySelector(".cssJs .txt_h1");
function colorChange(){
const currentColor = color.style.color; // 현재 이 태그의 상태를 말해줌
let newColor; // 비어있는 변수
if(currentColor === "blue"){
newColor = "tomato"; // currentColor가 블루면 newColor가 토마토로 변함
}else {
newColor = "blue";
}
color.style.color = newColor; // 비어있는 변수에 값을 넣어줌
}
color.addEventListener("click", colorChange);

className로 폰트 컬러 변경(className을 쓰면 모든 그 클래스네임을 가진 애들이 변경)

const h1 = document.querySelector(".h1_color");
function clickColor(){
const activeClass = "active";
if(h1.className === activeClass){
h1.className = "h1_color";
}else {
h1.className = activeClass;
}
};
h1.addEventListener("click", clickColor);

classList로 폰트 컬러 변경(classList는 class명들중에 contain이란 메소드를 써서 그 애들만 변경하는 것)

const h2 = document.querySelector(".classList");
function classList(){
const classListActive = "active";
if(h2.classList.contains(classListActive)){
h2.classList.remove(classListActive);
}else{
h2.classList.add(classListActive);
}
};
h2.addEventListener("click", classList);

toggle을 이용한 컬러변경

const h3 = document.querySelector(".toggle");
function toggleClass(){
h3.classList.toggle("active");
};
h3.addEventListener("click", toggleClass);
