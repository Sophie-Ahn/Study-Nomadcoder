자바스크립트는 html의 정보를 불러올 수 있음
html의 코드를 자바스크립트의 관점에서 보는 것. 그래서 시작할때 document를 이용함
document는 html를 불러오게 해줌

'title'이란 id안에 있는 값을 innterText를 이용해서 변경해줌
const title = document.getElementById("title");
title.innerText = "test word";

getElementById = id를 선택(id를 갖고 오는것을 알고 있기 때문에 따로 #을 안 넣어줘도 됨)
getElementsByClassName = class를 선택
querySelector = css selector로서 어떤 class나 id안에 태그를 가져올 수 있음(클래스나 id로 갖고올때 .이나 #을 써야함)
querySelectorAll = css selector로서 어떤 class나 id안에 태그를 가져올 수 있는데 해당되는 모든 것을 가져 올 수 있음
getElementsByTagName = div, h1, p 등등 태그 자체를 가져 오는 것

어떤 동작을 할때 event를 한다고 말할 수 있음

console.dir("h1");에서 앞에 on이라고 붙어 있는 것들은 다 event를 걸 수 있는 메소드

자바스크립트를 이용해서 컬러를 변경함

const exam1 = document.querySelector(".exam1_h1");
exam1.style.color = "#f00";

클릭을 했을때 이벤트가 뜨게

function clickEvent(){
console.log("클릭됨");
}
exam1.addEventListener("click", clickEvent);

마우스가 그 태그 안으로 들어갈 때

function mouseEnterEvent(){
console.log("마우스 움직임");
}
exam1.addEventListener("mouseenter", mouseEnterEvent);

태그에 event를 추가하는게 아니라 window에 event를 거는것

resize 하는 방법

function eventResize(){
document.body.style.backgroundColor = "beige";
};
window.addEventListener("resize", eventResize);
브라우저의 사이즈가 변경이 될때 event가 나오는것

};
